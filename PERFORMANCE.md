```

```

```javascript
const coupon = new CouponJS({
  verbose: true,
  logPerformance: true,
  maxNumberOfCouponsToGenerate: maxCoupon
});

const result = coupon.generate({
  characterSet: { builtIn: ['CHARSET_ALNUM'] },
  length: couponLength,
  numberOfCoupons: maxCoupon
});
```
